148status_t I2C::write( uint8_t address, const uint8_t *dp, int length, bool stop )
149{
150 status_t r = kStatus_Fail;
151 size_t txCount = 0xFFU;
152
153 if ( (r = LPI2C_MasterStart( unit_base, address, kLPI2C_Write )) )
154 return r;
155
156 LPI2C_MasterGetFifoCounts(unit_base, NULL, &txCount);
157 while (txCount)
158 {
159 LPI2C_MasterGetFifoCounts(unit_base, NULL, &txCount);
160 }
161
162 if ( LPI2C_MasterGetStatusFlags( unit_base ) & kLPI2C_MasterNackDetectFlag )
163 return kStatus_LPI2C_Nak;
164
165 if ( (r = LPI2C_MasterSend( unit_base, (uint8_t *)dp, length )) )
166 {
167 if ( r == kStatus_LPI2C_Nak )
168 LPI2C_MasterStop( unit_base );
169
170 return r;
171 }
172
173 if ( stop )
174 return LPI2C_MasterStop( unit_base );
175
176 return kStatus_Success;
177}

179status_t I2C::read( uint8_t address, uint8_t *dp, int length, bool stop )
180{
181 status_t r = kStatus_Fail;
182 size_t txCount = 0xFFU;
183
184 if ( (r = LPI2C_MasterStart( unit_base, address, kLPI2C_Read )) )
185 return r;
186
187 do {
188 LPI2C_MasterGetFifoCounts( unit_base, NULL, &txCount );
189 } while ( txCount );
190
191 if ( LPI2C_MasterGetStatusFlags( unit_base ) & kLPI2C_MasterNackDetectFlag )
192 return kStatus_LPI2C_Nak;
193
194 if ( (r = LPI2C_MasterReceive( unit_base, dp, length )) )
195 return r;
196
197 if ( stop )
198 return LPI2C_MasterStop( unit_base );
199
200 return kStatus_Success;
201}
